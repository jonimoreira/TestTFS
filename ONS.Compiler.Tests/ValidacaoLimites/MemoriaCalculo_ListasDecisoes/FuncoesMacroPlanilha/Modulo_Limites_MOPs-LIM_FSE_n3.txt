Public Function LIM_FSE_n3(xpercarga, xcargaSIN, xelocc, xhbo, xlimite_fse)
'********************************************
'IO-ON.SSE, REV. 61 - Vigência: 03/01/2014
'********************************************
lim = "inicial"

If xcargaSIN >= 63000 And xhbo >= 700 Then
    lim = xlimite_fse ' utiliza o limite normal, já que a limitação de RSE é suficiente

ElseIf xcargaSIN < 63000 Then
    Select Case xpercarga
        Case "PESADA", "MEDIA"
            If xhbo >= 500 Then
                lim = xlimite_fse ' utiliza o limite normal, já que a limitação de RSE é suficiente
            End If
        Case "LEVE", "MINIMA"
            lim = xlimite_fse ' utiliza o limite normal, já que a limitação de RSE é suficiente
    End Select

End If

If lim = "inicial" Then
    lim = 3600 'Valor adotado independentemente de período de carga e da geração Hbo. Item 6.3, b.3)
End If

LIM_FSE_n3 = lim

End Function
