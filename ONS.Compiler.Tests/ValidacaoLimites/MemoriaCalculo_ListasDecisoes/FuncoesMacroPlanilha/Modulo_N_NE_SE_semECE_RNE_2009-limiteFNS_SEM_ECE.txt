Public Function limiteFNS_SEM_ECE(xcarga_SIN, xMqIPU) As Variant

'************************************************
'IO-ON.NSE, rev 34.     Vigência: 04/10/2012
'MOP/CNOS 052/2014. Vigencia: 17/04/2014
'************************************************
Select Case xMqIPU
    Case 4
        Select Case xcarga_SIN
            Case Is >= 80000
                lim = 3100
            Case Is > 60000
                lim = 3100
            Case 53000 To 60000
                lim = 3100
            Case Is <= 53000
                lim = 3000
        End Select
    Case Is <= 3
        Select Case xcarga_SIN
            Case Is >= 80000
                lim = 3700
            Case Is > 60000
                lim = 3600
            Case 53000 To 60000
                lim = 3500
            Case Is <= 53000
                lim = 3500
        End Select
End Select

limiteFNS_SEM_ECE = lim

        
End Function
