Public Function Limite_RSE(xelo, xangra, xMaq, xGerIPU, xrsul, xpercarga)
'************************************************
'IO-ON.SSE,revisão 61, Vig. 03/01/2014.
'************************************************

Select Case xpercarga
    Case "PESADA", "MEDIA"
        Select Case xGerIPU
            Case Is <= 6700
                lim = 9100
            Case Is <= 7200
                lim = 8400
        End Select
'**************
    Case Is = "LEVE", "MINIMA"
        Select Case xGerIPU
            Case Is <= 6700
                lim = 9200
            Case Is <= 7200
                lim = 8500
        End Select
End Select

'Redução de limites de acordo com ANGRA I ou ANGRA II
Select Case xpercarga
    Case "LEVE", "MINIMA"
        Select Case xangra
            Case "0Ligada" ' Angra 1 e Angra 2 desligadas
                lim = lim - 400
            Case "1Ligada" ' Angra 1 ou Angra 2 desligada
                lim = lim - 200
            Case "2Ligadas"
                lim = lim - 0
        End Select
End Select

'Redução de limites de acordo com o ELO e ANGRA I ou ANGRA II
Select Case xpercarga
    Case "LEVE", "MINIMA"
        Select Case xangra
            Case "0Ligada" ' Angra 1 e Angra 2 desligadas
                lim = lim - 400
            Case "1Ligada" ' Angra 1 ou Angra 2 desligada
                lim = lim - 200
            Case "2Ligadas"
                lim = lim - 0
        End Select
End Select

Limite_RSE = lim

End Function
